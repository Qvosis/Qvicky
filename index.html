// ========================================== // ВСТАВЬ СВОИ ДАННЫЕ ОТ ZEGOCLOUD СЮДА: // ========================================== const ZEGOCLOUD_APP_ID = 1787230976; // ТВОЙ APP ID (Без кавычек) const ZEGOCLOUD_SERVER_SECRET = "13781fdd5faecce543c56060642fb0b2"; // ТВОЙ SERVER SECRET // --- ГЛОБАЛЬНЫЕ ПЕРЕМЕННЫЕ --- let myName = ""; let currentChatUser = ""; let zegoInstance = null; let incomingRoomID = ""; let incomingCallerName = ""; let mqttClient = null; let localDB = { contacts: JSON.parse(localStorage.getItem('qvicky_contacts')) || [], messages: JSON.parse(localStorage.getItem('qvicky_messages')) || {} }; function initApp() { let input = document.getElementById('my-username').value.trim(); if (!input) return alert("Введите юзернейм!"); // ПРАВКА 1: Принудительно делаем юзернейм строчным (маленькие буквы) myName = input.startsWith('@') ? input.toLowerCase() : '@' + input.toLowerCase(); document.getElementById('login-overlay').style.display = 'none'; document.getElementById('my-display-name').innerText = myName; renderContacts(); showScreen('contacts-screen'); connectToMQTT(); } // ========================================== // БАЗА ДАННЫХ И СЕТЬ (MQTT) // ========================================== function connectToMQTT() { const statusEl = document.getElementById('connection-status'); // ПРАВКА 2: Меняем сервер на более стабильный HiveMQ mqttClient = mqtt.connect('wss://broker.hivemq.com:8443/mqtt'); mqttClient.on('connect', () => { statusEl.innerText = "🟢 В сети (HiveMQ)"; statusEl.style.color = "#4caf50"; // Подписываемся на свои каналы (без символа @ для надежности в пути) const safeId = myName.replace('@', ''); mqttClient.subscribe(`qvicky/inbox/${safeId}`); mqttClient.subscribe(`qvicky/call/${safeId}`); }); mqttClient.on('error', (err) => { statusEl.innerText = "🔴 Ошибка подключения"; statusEl.style.color = "#f44336"; console.error("MQTT Error: ", err); }); // Обработка входящих mqttClient.on('message', (topic, message) => { try { const data = JSON.parse(message.toString()); const safeId = myName.replace('@', ''); if (topic === `qvicky/inbox/${safeId}`) { saveMessageLocally(data.sender, data.sender, data.text); if (currentChatUser === data.sender) renderChat(currentChatUser); } else if (topic === `qvicky/call/${safeId}`) { const alertBox = document.getElementById('incoming-call-alert'); if (data.status === 'ringing') { incomingRoomID = data.roomID; incomingCallerName = data.initiator; document.getElementById('caller-name').innerText = `Звонок от ${data.initiator}`; alertBox.style.display = 'flex'; } else if (data.status === 'ended') { alertBox.style.display = 'none'; if (zegoInstance) closeZegoUI(); } } } catch (e) { console.error("Ошибка парсинга:", e); } }); } // ========================================== // ZEGOCLOUD (ЛОГИКА ЗВОНКОВ) // ========================================== function startZegoCall(roomID) { const wrapper = document.getElementById('zego-wrapper'); wrapper.style.display = 'block'; const kitToken = ZegoUIKitPrebuilt.generateKitTokenForTest( ZEGOCLOUD_APP_ID, ZEGOCLOUD_SERVER_SECRET, roomID, myName, myName ); zegoInstance = ZegoUIKitPrebuilt.create(kitToken); zegoInstance.joinRoom({ container: document.getElementById('zego-container'), scenario: { mode: ZegoUIKitPrebuilt.OneONoneCall }, showPreJoinView: false, onLeaveRoom: () => { closeZegoUI(); const targetUser = currentChatUser || incomingCallerName; if (targetUser && mqttClient) { const targetSafeId = targetUser.replace('@', ''); mqttClient.publish(`qvicky/call/${targetSafeId}`, JSON.stringify({ status: 'ended' })); } } }); } function closeZegoUI() { document.getElementById('zego-wrapper').style.display = 'none'; if (zegoInstance) { zegoInstance.destroy(); zegoInstance = null; } } function startCall() { if (!currentChatUser) return; const roomID = 'room_' + myName.replace('@', ''); startZegoCall(roomID); if (mqttClient) { const targetSafeId = currentChatUser.replace('@', ''); mqttClient.publish(`qvicky/call/${targetSafeId}`, JSON.stringify({ initiator: myName, roomID: roomID, status: 'ringing' })); } } function answerCall() { document.getElementById('incoming-call-alert').style.display = 'none'; currentChatUser = incomingCallerName; startZegoCall(incomingRoomID); } function rejectCall() { document.getElementById('incoming-call-alert').style.display = 'none'; if (mqttClient) { const targetSafeId = incomingCallerName.replace('@', ''); mqttClient.publish(`qvicky/call/${targetSafeId}`, JSON.stringify({ status: 'ended' })); } } // ========================================== // ЛОГИКА ЧАТА И КОНТАКТОВ // ========================================== function saveMessageLocally(chatWith, sender, text) { if (!localDB.messages[chatWith]) localDB.messages[chatWith] = []; localDB.messages[chatWith].push({ sender: sender, text: text, time: new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}) }); localStorage.setItem('qvicky_messages', JSON.stringify(localDB.messages)); } function addContact() { let input = document.getElementById('search-input').value.trim(); if (!input) return; // ПРАВКА 3: Принудительно делаем юзернейм контакта строчным let targetUser = input.startsWith('@') ? input.toLowerCase() : '@' + input.toLowerCase(); if (targetUser === myName) return alert("Это ваш юзернейм!"); if (localDB.contacts.includes(targetUser)) return alert("Уже в контактах!"); localDB.contacts.push(targetUser); localStorage.setItem('qvicky_contacts', JSON.stringify(localDB.contacts)); document.getElementById('search-input').value = ''; renderContacts(); } function renderContacts() { const container = document.getElementById('contacts-container'); container.innerHTML = ''; localDB.contacts.forEach(user => { const isVerified = user === "@qvosis" ? ' verified' : ''; // Изменено на маленькие буквы для проверки const initial = user.charAt(1).toUpperCase(); const div = document.createElement('div'); div.className = 'contact-item'; div.onclick = () => openChat(user); div.innerHTML = `
${initial}
${user}
`; container.appendChild(div); }); } function showScreen(screenId) { document.querySelectorAll('.screen').forEach(s => s.classList.remove('active')); document.getElementById(screenId).classList.add('active'); } function openChat(user) { currentChatUser = user; const isVerified = user === "@qvosis" ? ' verified' : ''; document.getElementById('chat-header-name').innerHTML = `${user}`; renderChat(user); showScreen('chat-screen'); } function renderChat(user) { const chatArea = document.getElementById('chat-messages'); chatArea.innerHTML = ''; const history = localDB.messages[user] || []; history.forEach(msg => { const div = document.createElement('div'); div.className = `msg ${msg.sender === myName ? 'me' : 'other'}`; div.innerHTML = `${msg.text} ${msg.time}`; chatArea.appendChild(div); }); chatArea.scrollTop = chatArea.scrollHeight; } function sendMessage() { const input = document.getElementById('msg-input'); const text = input.value.trim(); if (!text || !currentChatUser) return; saveMessageLocally(currentChatUser, myName, text); renderChat(currentChatUser); input.value = ''; if (mqttClient) { const targetSafeId = currentChatUser.replace('@', ''); mqttClient.publish(`qvicky/inbox/${targetSafeId}`, JSON.stringify({ sender: myName, text: text })); } }
